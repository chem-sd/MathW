問題

以下の式を展開してください．

  • (1) \( ( 9 x - 8 y ) ( 7 x + 2 y ) \)
  • (2) \( ( 8 x + 5 y ) ( 5 x - 6 y ) \)
  • (3) \( ( 8 x + 4 y ) ( 4 x + 8 y ) \)
  • (4) \( ( 9 x - 2 y ) ( 9 x + 10 y ) \)
  • (5) \( ( 3 x - 6 y ) ( 7 x + 6 y ) \)
  • (6) \( ( 10 x - 2 y ) ( 3 x + 10 y ) \)
  • (7) \( ( 9 x - 7 y ) ( 9 x - 4 y ) \)
  • (8) \( ( 7 x + 5 y ) ( 2 x - y ) \)
  • (9) \( ( x + 4 y ) ( 5 x + 3 y ) \)
  • (10) \( ( 4 x + 9 y ) ( 10 x + 6 y ) \)
  • (1) \begin{align} ( 9 x - 8 y ) ( 7 x + 2 y ) &= 9x \times 7x + 9x \times 2y -8y \times 7x -8y \times 2y \\ &= 63x^2 + 18xy - 56xy -16y^2 \\ &= \underline{63x^2+-38xy-16y^2} \end{align}
  • (2) \begin{align} ( 8 x + 5 y ) ( 5 x - 6 y ) &= 8x \times 5x + 8x \times (-6y) +5y \times 5x + 5y \times (-6y) \\ &= 40x^2 -48xy +25xy -30y^2 \\ &= \underline{40x^2-23xy-30y^2} \end{align}
  • (3) \begin{align} ( 8 x + 4 y ) ( 4 x + 8 y ) &= 8x \times 4x + 8x \times 8y +4y \times 4x +4y \times 8y \\ &= 32x^2 + 64xy + 16xy + 32y^2 \\ &= \underline{32x^2+80xy+32y^2} \end{align}
  • (4) \begin{align} ( 9 x - 2 y ) ( 9 x + 10 y )&= 9x \times 9x + 9x \times 10y -2y \times 9y - 2y \times 10y \\ &= 81x^2 + 90xy -18xy -20y^2 \\ &= \underline{81x^2+72xy-20y^2} \end{align}
  • (5) \begin{align} ( 3 x - 6 y ) ( 7 x + 6 y ) &= 3x \times 7x + 3x \times 6y - 6y \times 7x - 6y \times 6y \\ &= 21x^2 +18xy - 42xy -36y^2 \\ &= \underline{21x^2-24xy-36y^2} \end{align}
  • (6) \begin{align} ( 10 x - 2 y ) ( 3 x + 10 y ) &= 10x \times 3x + 10x \times 10y -2y \times 3x - 2y \times 10y \\ &= 30x^2 +100xy -6xy - 20y^2 \\ &= \underline{30x^2+94xy-20y^2} \end{align}
  • (7) \begin{align} ( 9 x - 7 y ) ( 9 x - 4 y ) &= 9x \times 9x + 9x \times (-4y) - 7y \times 9x -7y \times (-4y) \\ &= 81x^2 -36xy -63xy +28y^2 \\ &= \underline{81x^2+-99xy+28y^2} \end{align}
  • (8) \begin{align} ( 7 x + 5 y ) ( 2 x - y ) &= 7x \times 2x + 7x \times (-y) + 5y \times 2x +5y \times (-y) \\ &= 14x^2 -7xy + 10xy -5y^2 \\ &= \underline{14x^2+3xy-5y^2} \end{align}
  • (9) \begin{align} ( x + 4 y ) ( 5 x + 3 y ) &= x \times 5x + x \times 3y + 4y \times 5x + 4y \times 3y \\ &= 5x^2 +3xy +20xy +12y^2 \\ &= \underline{5x^2+23xy+12y^2} \end{align}
  • (10) \begin{align} ( 4 x + 9 y ) ( 10 x + 6 y ) &= 4x \times 10x + 4x \times 6y + 9y \times 10x +9x \times 6y \\ &= 40x^2 +24xy +90xy + 54y^2 \\ &= \underline{40x^2+114xy+54y^2} \end{align}
