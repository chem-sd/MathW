問題

以下の式を展開してください．

  • (1) \(　( x + 9 ) ( x + 10 ) \)
  • (2) \(　( x - 7 ) ( x - 9 ) \)
  • (3) \(　( x - 5 ) ( x + 9 ) \)
  • (4) \(　( x - 6 y ) ( x - y ) \)
  • (5) \(　( x + 6 y ) ( x - 7 y ) \)
  • (6) \(　( x - 3 ) ( x - 3 ) \)
  • (7) \(　( x + 6 y ) ( x - 5 y ) \)
  • (8) \(　( x - 1 ) ( x + 9 ) \)
  • (9) \(　( x + 6 y ) ( x - 7 y ) \)
  • (10) \(　( x - 4 y ) ( x + 8 y ) \)
  • (1) \begin{align} ( x + 9 ) ( x + 10 ) &= x^2 + (9+10)x + 9 \times 10 \\ &= \underline{x^2+19x+90} \end{align}
  • (2) \begin{align} ( x - 7 ) ( x - 9 ) &= x^2 + (-7-9)x + (-7) \times (-9) \\ &= \underline{x^2-16x+63} \end{align}
  • (3) \begin{align} ( x - 5 ) ( x + 9 ) &= x^2 + (-5+9)x + (-5) \times 9 \\ &= \underline{x^2+4x-45} \end{align}
  • (4) \begin{align} ( x - 6 y ) ( x - y ) &= x^2 + (-6y-y)x + (-6y) \times (-y) \\ &= \underline{x^2-7xy+6y^2} \end{align}
  • (5) \begin{align} ( x + 6 y ) ( x - 7 y ) &= x^2 + (6y-7y)x + 6y \times (-7y) \\ &= \underline{x^2-xy-42y^2} \end{align}
  • (6) \begin{align} ( x - 3 ) ( x - 4 ) &= x^2 + (-3-4)x + (-3) \times (-4) \\ &= \underline{x^2-7x+12} \end{align}
  • (7) \begin{align} ( x + 6 y ) ( x - 5 y ) &= x^2 + (6y-5y)x + 6y \times (-5y) \\ &= \underline{x^2+xy-30y^2} \end{align}
  • (8) \begin{align} ( x - 1 ) ( x + 9 ) &= x^2 + (-1+9)x + (-1) \times 9 \\ &= \underline{x^2+8x-9} \end{align}
  • (9) \begin{align} ( x + 6 y ) ( x - 3 y ) &= x^2 + (6y-3y)x + 6y \times (-3y) \\ &= \underline{x^2+3xy-18y^2} \end{align}
  • (10) \begin{align} ( x - 4 y ) ( x + 8 y ) &= x^2 + (-4y+8y) + (-4y) \times 8y \\ &= \underline{x^2+4xy-32y^2} \end{align}
