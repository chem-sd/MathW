問題

以下の式を展開してください．

  • (1) \(　( 2 x - 4 y ) ( 2 x - 8 y ) \)
  • (2) \(　( x + 6 y ) ( x - 6 y ) \)
  • (3) \(　( 8 x - 9 y ) ^2　\)
  • (4) \(　( 4 x + 3 y ) ( 4 - 3 y ) \)
  • (5) \(　( 2 x - 2 y ) ( 9 x - 4 y ) \)
  • (6) \(　( 7 x + 8 y ) ^2 \)
  • (7) \(　( 2 x + 4 y ) ^2　\)
  • (8) \(　( 4 x - 4 y ) ^2) \)
  • (9) \(　( 6 x - 5 y ) ( 1 x + 8 y )　\)
  • (10) \(　( 6 x + 9 y ) ^2 \)
  • (1) \begin{align} ( 2 x - 4 y ) ( 2 x - 8 y ) &= 2x \times 2x + 2x \times (-8y) + (-4y) \times 2x + (-4y) \times (-8y) \\ &= 4x^2-16xy-8xy+32y^2 \\ &= \underline{4x^2-24xy+32y^2} \end{align}　
  • (2) \begin{align} ( x + 6 y ) ( x - 6 y ) &= x^2 - (6y)^2 \\ &= \underline{x^2-36y^2} \end{align}　
  • (3) \begin{align} ( 8 x - 9 y ) ^2 &= (8x)^2 - 2 \times 8x \times 9y + (9y)^2 \\ &= \underline{64x^2 - 144xy+81y^2} \end{align}　
  • (4) \begin{align} ( 4 x + 3 y ) ( 4 - 3 y ) &= (4x)^2 - (3y)^2 \\ &= \underline{16x^2-9y^2} \end{align}　
  • (5) \begin{align} ( 2 x - 2 y ) ( 9 x - 4 y ) &= 2x \times 9x + 2x \times (-4y) + (-2y) \times 9x + (-2y) \times (-4y) \\ &= 18x^2 -8xy -18xy +8y^2 \\ &= \underline{18x^2-26xy+8y^2} \end{align}
  • (6) \begin{align} ( 7 x + 8 y ) ^2 &= (7x)^2 + 2 \times 7x \times 8y + (8y)^2 \\ &= \underline{49x^2+112xy+64y^2} \end{align}
  • (7) \begin{align} ( 2 x + 4 y ) ^2 &= (2x)^2 + 2 \times 2x \times 4y +(4y)^2 \\ &= \underline{4x^2+16xy+16y^2} \end{align}
  • (8) \begin{align} ( 4 x - 4 y ) ^2 &= (4x)^2 - 2 \times 4x \times (4y) + (4y)^2 \\ &= \underline{16x^2-32xy+16y^2} \end{align}
  • (9) \begin{align} ( 6 x - 5 y ) ( x + 8 y ) &= 6x \times x + 6x \times 8y + (-5y) \times x + (-5y) \times 8y \\ &= 6x^2+48xy-5xy-40y^2 \\ &= \underline{6x^2+43xy-40y^2} \end{align}
  • (10) \begin{align} ( 6 x + 9 y ) ^2 &= (6x)^2 + 2 \times 6x \times 9y + (9y)^2 \\ &= \underline{36x^2+108xy+81y^2} \end{align}
