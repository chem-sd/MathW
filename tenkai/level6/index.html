問題

以下の式を展開してください．

  • (1) \(　( 9 x + 6 y ) ( 9 x - 6 y ) \)
  • (2) \(　( 9 x + 4 y ) ( 9 x - 4 y ) \)
  • (3) \(　( 9 x + y ) ( 9 x - y )\)
  • (4) \(　( 5 x + y ) ( 5 x - y ) \)
  • (5) \(　( 4 x + 5 y ) ( 4 x - 5 y ) \)
  • (6) \(　( 10 x + 8 y ) ( 10 x - 8 y ) \)
  • (7) \(　( x + 3 y ) ( x - 3 y )\)
  • (8) \(　( 6 x + 10 y ) ( 6 x - 10 y ) \)
  • (9) \(　( 2 x + 5 y ) ( 2 x - 5 y )\)
  • (10) \(　( 7 x + y ) ( 7 x - y ) \)
  • (1) \begin{align} ( 9 x + 6 y ) ( 9 x - 6 y ) &= (9x)^2 - (6y)^2 \\ &= \underline{81x^2-36y^2} \end{align}
  • (2) \begin{align} ( 9 x + 4 y ) ( 9 x - 4 y ) &= (9x)^2 - (4y)^2 \\ &= \underline{81x^2-16y^2} \end{align}
  • (3) \begin{align} ( 9 x + y ) ( 9 x - y ) &= (9x)^2 - y^2 \\ &= \underline{81x^2-y^2} \end{align}
  • (4) \begin{align} ( 5 x + y ) ( 5 x - y ) &= (5x)^2 - y^2 \\ &= \underline{25x^2-y^2} \end{align}
  • (5) \begin{align} ( 4 x + 5 y ) ( 4 x - 5 y ) &= (4x)^2 - (5y)^2 \\ &= \underline{16x^2-25y^2} \end{align}
  • (6) \begin{align} ( 10 x + 8 y ) ( 10 x - 8 y ) &= (10x)^2 - (8y)^2 \\ &= \underline{100x^2-64y^2} \end{align}
  • (7) \begin{align} ( x + 3 y ) ( x - 3 y ) &= x^2 - (3y)^2 \\ &= \underline{x^2-9y^2} \end{align}
  • (8) \begin{align} ( 6 x + 10 y ) ( 6 x - 10 y ) &= (6x)^2 - (10y)^2 \\ &= \underline{36x^2-100y^2} \end{align}
  • (9) \begin{align} ( 2 x + 5 y ) ( 2 x - 5 y ) &= (2x)^2 - (5y)^2 \\ &= \underline{4x^2-5y^2} \end{align}
  • (10) \begin{align} ( 7 x + y ) ( 7 x - y ) &= (7x)^2 - y^2 \\ &= \underline{49x^2-y^2} \end{align}
