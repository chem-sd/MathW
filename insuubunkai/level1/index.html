  • (5) \begin{align} -3x-6y = \underline{-3(x+2)} \end{align}
  • (6) \begin{align} 100x+300y = \underline{100(x+3y)} \end{align}
  • (7) \begin{align} 4x-12y = \underline{4(x-3y)} \end{align}
  • (8) \begin{align} -3x+9y = \underline{-3(x-3y)} \end{align}
  • (9) \begin{align} 144x+24y = \underline{24(6x+y)} \end{align}
